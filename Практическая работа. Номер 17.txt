using System;

namespace Практическая_работа__1
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Введите число");
            int t = Int32.Parse(Console.ReadLine());
            if (t > 60)
            {
                Console.WriteLine("Пожарная ситуация");
                Console.ReadKey();
            }
            if (t <= 60)
            {
                Console.WriteLine("Ситуация в норме");
                Console.ReadKey();
            }
        }
    }
}
